#include<iostream>
#include<vector>
using namespace std;
int main(){
    int n;
    if(!(cin>>n)) return 0;
    vector<int>sticks(n);
    for(int i=0;i<n;i++){
        cin>>sticks[i];
    }
    int sum=0;
    for(int i=n-1;i>=0;i-=2){
        sum+=sticks[i];
    }
    cout<<sum;
    return 0;
}
